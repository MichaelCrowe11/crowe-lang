See It In Action

// Real-time chat with AI sentiment analysis
store ChatStore {
  state messages: Message[] = [];
  stream liveChat: Message = "wss://chat.example.com";
  ai sentiment: Classifier = model("sentiment-analysis");
  
  action sendMessage(text: string) {
    messages = [...messages, { text, user: currentUser }];
    analyzeSentiment(text);
  }
  
  action async analyzeSentiment(text: string) {
    const result = await sentiment.classify(text);
    // Update message with sentiment
  }
}

component ChatApp() {
  state inputText: string = "";
  computed store: any { return useChatStore(); }
  
  effect onMount {
    store.connectLiveChat();
  }
  
  render {
    <div className="chat-app">
      {store.messages.map(msg => (
        <MessageBubble key={msg.id} message={msg} />
      ))}
      <input 
        value={inputText} 
        onChange={(e) => inputText = e.target.value}
        onKeyPress={(e) => e.key === 'Enter' && store.sendMessage(inputText)}
      />
    </div>
  }
}

Getting Started

# Install globally
npm install -g crowe-lang

# Create a new project
crowe init my-app
cd my-app

# Start development
npm run dev

# Or compile a single file
crowe compile app.crowe -o app.tsx --watch

Search for "Crowe Language Support" in VS Code extensions for syntax highlighting, IntelliSense, and integrated compilation.
